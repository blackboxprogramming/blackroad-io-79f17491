trinary_logic.py 
class TrinaryState:
    """Lucidia trinary logic: 1, 0, -1 states"""

    TRUE = 1
    FALSE = 0
    UNKNOWN = -1

    @classmethod
    def evaluate(cls, condition):
        if condition is None:
            return cls.UNKNOWN
        return cls.TRUE if condition else cls.FALSE

    @classmethod
    def combine(cls, *states):
        """Combine multiple trinary states"""
        if cls.UNKNOWN in states:
            return cls.UNKNOWN
        return cls.TRUE if all(s == cls.TRUE for s in states) else cls.FALSE
