Deploy an agent in minutes

Simple primitives for complex behaviors. Define capabilities, connect to the event bus, and let the orchestration layer handle coordination.

Read the Docs 
# Initialize agent with memory
from blackroad import Agent, Memory

agent = Agent(
    name="lucidia-7",
    memory=Memory(
        backend="ps-sha-inf",
        journal="append-only"
    )
)

# Register capabilities
agent.register("reasoning", trinary=True)
agent.register("contradiction-handling")

# Connect to event bus
await agent.connect(bus="nats://cluster")
await agent.start()
